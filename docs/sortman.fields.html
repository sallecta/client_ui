Start Sortman and enable drag sorting later by click

client_ui.run = function()
{
	el=document.querySelector('h1');el.innerHTML=el.innerHTML+' '+client_ui.version;
	
	//const inst = client_ui.mdls.sortman.run('.field',{disabled:true});
	const inst = client_ui.mdls.sortman.run('.field');
	el_drag_ctrl = document.querySelector('#drag_ctrl');
			console.log('inst.ev_listeners_status',inst.ev_listeners_status);
	if ( inst.ev_listeners_status == '-' )
	{
		el_drag_ctrl.innerHTML='Enable';
	}
	else if ( inst.ev_listeners_status == '+' )
	{
		el_drag_ctrl.innerHTML='Disable';
	}
	drag_ctrl.addEventListener('click', function(a_ev)
		{
			const app=client_ui.mdls.sortman;
			if ( inst.ev_listeners_status == '-' )
			{
				client_ui.mdls.sortman.ev_listeners(inst,'add');
				a_ev.target.innerHTML='Disable';
			}
			else if ( inst.ev_listeners_status == '+' )
			{
				client_ui.mdls.sortman.ev_listeners(inst,'remove');
				a_ev.target.innerHTML='Enable';
			}
			a_ev.preventDefault();
		}
	);
}
